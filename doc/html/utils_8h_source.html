99  return std::to_string(std::chrono::duration_cast<std::chrono::duration<double, std::micro>>(d).count()) + " us";
100  } else if (d < std::chrono::seconds(1)) {
101  return std::to_string(std::chrono::duration_cast<std::chrono::duration<double, std::milli>>(d).count()) + " ms";
102  } else {
103  return std::to_string(print_dt(d)) + " s";
