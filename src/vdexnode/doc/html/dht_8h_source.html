310  /* When performing a search, we search for up to SEARCH_NODES closest nodes
311  to the destination, and use the additional ones to backtrack if any of
312  the target 8 turn out to be dead. */
313  static constexpr unsigned SEARCH_NODES {14};
314 
315  /* The number of bad nodes is limited in order to help determine
316  * presence of connectivity changes. See
317  * https://github.com/savoirfairelinux/opendht/issues/137 for details.
318  *

320  * the network is small, normal search expiration process will handle the
321  * situation.
322  * */
