86  time_point run() {
87  syncTime();
88  while (not timers.empty()) {
89  auto timer = timers.begin();
90  /*
91  * Running jobs scheduled before "now" prevents run+rescheduling
92  * loops before this method ends. It is garanteed by the fact that a
93  * job will at least be scheduled for "now" and not before.
94  */
95  if (timer->first > now)
96  break;
97 
98  auto job = std::move(timer->second);
99  timers.erase(timer);
100 
101  if (job->do_)
102  job->do_();
103  }
104  return getNextJobTime();
105  }
106 
107  inline time_point getNextJobTime() const {
108  return timers.empty() ? time_point::max() : timers.begin()->first;
109  }

115  inline const time_point& time() const { return now; }
116  inline time_point syncTime() { return (now = clock::now()); }
