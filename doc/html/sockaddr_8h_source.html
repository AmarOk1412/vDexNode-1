257  struct ipCmp {
258  bool operator()(const SockAddr& a, const SockAddr& b) const {
259  if (a.len != b.len)
260  return a.len < b.len;
261  socklen_t start, len;
262  switch(a.getFamily()) {
263  case AF_INET:
264  start = offsetof(sockaddr_in, sin_addr);
265  len = sizeof(in_addr);
266  break;
267  case AF_INET6:
268  start = offsetof(sockaddr_in6, sin6_addr);
269  // don't consider more than 64 bits (IPv6)
270  len = 8;
271  break;
272  default:
273  start = 0;
274  len = a.len;
275  break;
276  }
277  return std::memcmp((uint8_t*)a.get()+start,
278  (uint8_t*)b.get()+start, len) < 0;
279  }
280  };
